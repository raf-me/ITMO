# --- 2. Рекурсивный список файлов в lab0 с "wo", сортировка по жестким ссылкам (убывание) ---
# find: находит файлы рекурсивно
# -type f: ищет только файлы
# -print0: выводит имена файлов, разделенные нулевым символом (безопасно для имен с пробелами)
# xargs -0: обрабатывает список файлов из find
# grep -l "wo": ищет файлы, содержащие "wo", -l выводит только имена файлов
# ls -l: выводит информацию о файлах
# sort -nrk 3: сортирует по 3-му полю (количество жестких ссылок), числово, убывание

find lab0 -type f -print0 | xargs -0 grep -l "wo" | xargs -0 ls -l | sort -nrk 3


# --- 3. Рекурсивное выведение содержимого файлов из lab0, начинающихся на 's', сортировка z->a, вывод ошибок ---
# find: находит файлы, начинающиеся на 's' рекурсивно
# xargs -I {} cat {}: выполняет cat для каждого найденного файла
# sort -r: сортирует строки в обратном алфавитном порядке (z->a)
# 2>&1: перенаправляет stderr в stdout

find lab0 -type f -name "s*" 2>&1 | xargs -I {} cat {} 2>&1 | sort -r

-lruR - 
`-lru`: Эта опция используется с некоторыми командами, такими как `free` и `slabtop`, для вывода информации о кэше страниц в порядке уменьшения времени последнего использования (Least Recently Used). Другими словами, сначала выводятся страницы, которые использовались дольше всего. 
 `-R`: Эта опция часто используется в сочетании с `-lru` для указания на то, что информация о кэше должна быть выведена рекурсивно. Это означает, что будет предоставлена информация не только о кэше верхнего уровня, но и о кэшах, вложенных в него.
